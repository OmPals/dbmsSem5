	 declare j int;
	 declare i int;
BEGIN
	j:=1;
	i:=1;
    FOR r IN (select pres_id from prescription order by pres_id asc)
    LOOP
		for cn in 1..3 loop
			if j%100 = 0 then
				insert into yields values(r.pres_id,100) ;
				j=j+1;
			else
				insert into yields values(r.pres_id,j%100);
				j=j+1;
			end if;   
		end loop;
		i=i+1;
	END LOOP;
